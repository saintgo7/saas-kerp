# 4대보험 EDI 연동 설계서

## 1. 개요

### 1.1 4대보험이란?
한국의 사회보험 체계로, 모든 사업장에서 의무적으로 가입해야 하는 4가지 보험입니다.

```
┌─────────────────────────────────────────────────────────────────┐
│                        4대보험 체계                              │
├─────────────────────────────────────────────────────────────────┤
│                                                                  │
│  ┌─────────────────┐  ┌─────────────────┐                       │
│  │    국민연금     │  │    건강보험     │                       │
│  │                 │  │                 │                       │
│  │ 관리: 국민연금  │  │ 관리: 건강보험  │                       │
│  │       공단      │  │       공단      │                       │
│  │                 │  │                 │                       │
│  │ 요율: 9%       │  │ 요율: 7.09%    │                       │
│  │ (사업주 4.5%)  │  │ (사업주 3.545%)│                       │
│  └─────────────────┘  └─────────────────┘                       │
│                                                                  │
│  ┌─────────────────┐  ┌─────────────────┐                       │
│  │    고용보험     │  │    산재보험     │                       │
│  │                 │  │                 │                       │
│  │ 관리: 근로복지  │  │ 관리: 근로복지  │                       │
│  │       공단      │  │       공단      │                       │
│  │                 │  │                 │                       │
│  │ 요율: 1.8%     │  │ 요율: 업종별   │                       │
│  │ (사업주 0.9%)  │  │ (사업주 전액)  │                       │
│  └─────────────────┘  └─────────────────┘                       │
│                                                                  │
│  ※ 요율은 2024년 기준, 매년 변경될 수 있음                      │
│                                                                  │
└─────────────────────────────────────────────────────────────────┘
```

### 1.2 연동 필요 업무

| 업무 | 주기 | 신고 기관 |
|------|------|----------|
| **자격취득 신고** | 입사 시 | 4대보험 통합 |
| **자격상실 신고** | 퇴사 시 | 4대보험 통합 |
| **보수총액 신고** | 매년 3월 | 건강보험공단 |
| **월별 보험료 납부** | 매월 10일 | 각 공단 |
| **정산/연말정산** | 매년 | 각 공단 |

### 4.2 보험료 계산 코드 (Go)

```go
// internal/service/insurance/premium_calculator.go

package insurance

import (
    "math"
)

// InsuranceRates 2024년 4대보험 요율
type InsuranceRates struct {
    NationalPension   PensionRates
    HealthInsurance   BasicRates
    LongTermCare      LongTermCareRates
    EmploymentInsurance EmploymentRates
}

type PensionRates struct {
    Total     float64
    Employee  float64
    Employer  float64
    MinIncome int64 // 기준소득월액 하한
    MaxIncome int64 // 기준소득월액 상한
}

type BasicRates struct {
    Total    float64
    Employee float64
    Employer float64
}

type LongTermCareRates struct {
    Rate     float64 // 건강보험료의 비율
    Employee float64
    Employer float64
}

type EmploymentRates struct {
    Employee       float64
    EmployerRates  map[string]float64 // 기업 규모별
}

// DefaultRates 2024년 기본 요율
var DefaultRates = InsuranceRates{
    NationalPension: PensionRates{
        Total:     0.09,
        Employee:  0.045,
        Employer:  0.045,
        MinIncome: 370000,
        MaxIncome: 5900000,
    },
    HealthInsurance: BasicRates{
        Total:    0.0709,
        Employee: 0.03545,
        Employer: 0.03545,
    },
    LongTermCare: LongTermCareRates{
        Rate:     0.1295, // 건강보험료의 12.95%
        Employee: 0.5,
        Employer: 0.5,
    },
    EmploymentInsurance: EmploymentRates{
        Employee: 0.009,
        EmployerRates: map[string]float64{
            "under_150":        0.0025,
            "priority_150_plus": 0.0045,
            "150_to_1000":      0.0065,
            "over_1000":        0.0085,
        },
    },
}

type PremiumCalculator struct {
    rates InsuranceRates
}

func NewPremiumCalculator() *PremiumCalculator {
    return &PremiumCalculator{rates: DefaultRates}
}

// PremiumResult 4대보험료 계산 결과
type PremiumResult struct {
    MonthlyIncome       int64
    NationalPension     InsurancePremium
    HealthInsurance     InsurancePremium
    LongTermCare        InsurancePremium
    EmploymentInsurance InsurancePremium
    Total               InsurancePremium
}

type InsurancePremium struct {
    Employee int64
    Employer int64
}

// Calculate 4대보험료 계산
func (c *PremiumCalculator) Calculate(monthlyIncome int64, companySize string) *PremiumResult {
    // 국민연금 기준소득월액 적용
    pensionIncome := c.getPensionBaseIncome(monthlyIncome)

    // 국민연금
    nationalPension := InsurancePremium{
        Employee: c.roundToTens(float64(pensionIncome) * c.rates.NationalPension.Employee),
        Employer: c.roundToTens(float64(pensionIncome) * c.rates.NationalPension.Employer),
    }

    // 건강보험
    healthInsurance := InsurancePremium{
        Employee: c.roundToTens(float64(monthlyIncome) * c.rates.HealthInsurance.Employee),
        Employer: c.roundToTens(float64(monthlyIncome) * c.rates.HealthInsurance.Employer),
    }

    // 장기요양보험 (건강보험료 기준)
    healthTotal := healthInsurance.Employee + healthInsurance.Employer
    longTermCareTotal := c.roundToTens(float64(healthTotal) * c.rates.LongTermCare.Rate)
    longTermCare := InsurancePremium{
        Employee: c.roundToTens(float64(longTermCareTotal) * c.rates.LongTermCare.Employee),
        Employer: c.roundToTens(float64(longTermCareTotal) * c.rates.LongTermCare.Employer),
    }

    // 고용보험
    employerRate := c.rates.EmploymentInsurance.EmployerRates[companySize]
    if employerRate == 0 {
        employerRate = c.rates.EmploymentInsurance.EmployerRates["under_150"]
    }
    employmentInsurance := InsurancePremium{
        Employee: c.roundToTens(float64(monthlyIncome) * c.rates.EmploymentInsurance.Employee),
        Employer: c.roundToTens(float64(monthlyIncome) * employerRate),
    }

    return &PremiumResult{
        MonthlyIncome:       monthlyIncome,
        NationalPension:     nationalPension,
        HealthInsurance:     healthInsurance,
        LongTermCare:        longTermCare,
        EmploymentInsurance: employmentInsurance,
        Total: InsurancePremium{
            Employee: nationalPension.Employee + healthInsurance.Employee +
                      longTermCare.Employee + employmentInsurance.Employee,
            Employer: nationalPension.Employer + healthInsurance.Employer +
                      longTermCare.Employer + employmentInsurance.Employer,
        },
    }
}

// getPensionBaseIncome 국민연금 기준소득월액 적용
func (c *PremiumCalculator) getPensionBaseIncome(income int64) int64 {
    minIncome := c.rates.NationalPension.MinIncome
    maxIncome := c.rates.NationalPension.MaxIncome

    if income < minIncome {
        return minIncome
    }
    if income > maxIncome {
        return maxIncome
    }
    return income
}

// roundToTens 10원 단위 반올림
func (c *PremiumCalculator) roundToTens(amount float64) int64 {
    return int64(math.Round(amount/10) * 10)
}
```

### 5.2 EDI 파일 생성 코드 (Go)

```go
// internal/service/insurance/edi_generator.go

package insurance

import (
    "bytes"
    "fmt"
    "strings"
    "time"

    "golang.org/x/text/encoding/korean"
    "golang.org/x/text/transform"
    "k-erp/internal/domain"
)

const (
    lineEnding = "\r\n"
    fieldSep   = "|"
)

type EDIGenerator struct{}

func NewEDIGenerator() *EDIGenerator {
    return &EDIGenerator{}
}

// GenerateAcquisitionReport 자격취득 신고 파일 생성
func (g *EDIGenerator) GenerateAcquisitionReport(
    businessNumber string,
    employees []*domain.Employee,
    reportDate time.Time,
) ([]byte, error) {
    var buf bytes.Buffer

    // 헤더 레코드
    buf.WriteString(g.buildHeader(businessNumber, reportDate, len(employees)))

    // 데이터 레코드
    var totalIncome int64
    for _, emp := range employees {
        buf.WriteString(g.buildAcquisitionRecord(businessNumber, emp))
        totalIncome += emp.MonthlyIncome
    }

    // 트레일러 레코드
    buf.WriteString(g.buildTrailer(len(employees), totalIncome))

    // EUC-KR 인코딩 변환
    return g.encodeToEUCKR(buf.String())
}

// GenerateLossReport 자격상실 신고 파일 생성
func (g *EDIGenerator) GenerateLossReport(
    businessNumber string,
    employees []*domain.Employee,
    reportDate time.Time,
) ([]byte, error) {
    var buf bytes.Buffer

    // 헤더 레코드
    buf.WriteString(g.buildHeader(businessNumber, reportDate, len(employees)))

    // 데이터 레코드
    for _, emp := range employees {
        buf.WriteString(g.buildLossRecord(businessNumber, emp))
    }

    // 트레일러 레코드
    buf.WriteString(g.buildTrailer(len(employees), 0))

    // EUC-KR 인코딩 변환
    return g.encodeToEUCKR(buf.String())
}

func (g *EDIGenerator) buildHeader(businessNumber string, reportDate time.Time, count int) string {
    fields := []string{
        "H",                                    // 레코드 구분
        businessNumber,                         // 사업장관리번호
        reportDate.Format("20060102"),          // 신고일자 (YYYYMMDD)
        fmt.Sprintf("%05d", count),             // 총건수
        strings.Repeat(" ", 20),                // 담당자명 (공백)
    }
    return strings.Join(fields, fieldSep) + lineEnding
}

func (g *EDIGenerator) buildAcquisitionRecord(businessNumber string, emp *domain.Employee) string {
    foreignerCode := "1" // 내국인
    if emp.IsForeigner {
        foreignerCode = "2"
    }

    disabledCode := "1" // 일반
    if emp.IsDisabled {
        disabledCode = "2"
    }

    fields := []string{
        "D",                                           // 레코드 구분
        businessNumber,                                // 사업장관리번호
        strings.ReplaceAll(emp.ResidentNumber, "-", ""), // 주민등록번호
        g.padKorean(emp.Name, 20),                     // 성명
        emp.HireDate.Format("20060102"),               // 자격취득일
        "1",                                           // 취득구분 (당연취득)
        fmt.Sprintf("%09d", emp.MonthlyIncome),        // 보수월액
        "1",                                           // 근무형태 (상용)
        foreignerCode,                                 // 외국인여부
        disabledCode,                                  // 장애인여부
    }
    return strings.Join(fields, fieldSep) + lineEnding
}

func (g *EDIGenerator) buildLossRecord(businessNumber string, emp *domain.Employee) string {
    lossReasonCode := emp.LossReasonCode
    if lossReasonCode == "" {
        lossReasonCode = "11" // 퇴직
    }

    terminationDate := emp.TerminationDate
    if terminationDate == nil {
        terminationDate = &time.Time{}
    }

    fields := []string{
        "D",                                           // 레코드 구분
        businessNumber,                                // 사업장관리번호
        strings.ReplaceAll(emp.ResidentNumber, "-", ""), // 주민등록번호
        g.padKorean(emp.Name, 20),                     // 성명
        terminationDate.Format("20060102"),            // 자격상실일
        lossReasonCode,                                // 상실사유코드
    }
    return strings.Join(fields, fieldSep) + lineEnding
}

func (g *EDIGenerator) buildTrailer(count int, totalIncome int64) string {
    fields := []string{
        "T",                                    // 레코드 구분
        fmt.Sprintf("%05d", count),             // 총 레코드수
        fmt.Sprintf("%015d", totalIncome),      // 보수월액 합계
    }
    return strings.Join(fields, fieldSep) + lineEnding
}

// padKorean 한글 문자열을 EUC-KR 바이트 기준으로 패딩
func (g *EDIGenerator) padKorean(str string, length int) string {
    // EUC-KR로 변환하여 바이트 길이 계산
    eucStr, _ := g.encodeToEUCKR(str)
    currentLen := len(eucStr)

    if currentLen >= length {
        return str[:length/3] // 대략적인 자르기 (UTF-8 기준)
    }

    // 공백으로 패딩
    return str + strings.Repeat(" ", (length-currentLen))
}

// encodeToEUCKR UTF-8을 EUC-KR로 변환
func (g *EDIGenerator) encodeToEUCKR(s string) ([]byte, error) {
    reader := transform.NewReader(
        strings.NewReader(s),
        korean.EUCKR.NewEncoder(),
    )
    var buf bytes.Buffer
    if _, err := buf.ReadFrom(reader); err != nil {
        return nil, fmt.Errorf("EUC-KR 인코딩 실패: %w", err)
    }
    return buf.Bytes(), nil
}
```
